```python
# SPDX-License-Identifier: Apache-2.0
import json

import pytest
from typer.testing import CliRunner

from cairn.cli import app
from cairn.hosts import get_host

runner = CliRunner()

_MCP_HOSTS = ["cursor", "claude-desktop", "vscode", "gemini"]


@pytest.mark.parametrize("host_id", _MCP_HOSTS)
def test_install_writes_mcp_config(host_id, tmp_path, monkeypatch):
    monkeypatch.setenv("HOME", str(tmp_path))
    # Some host paths are derived at import time from sys.platform; that's fine —
    # they still expand against $HOME. Resolve the path the same way the host does.
    host = get_host(host_id)
    vault = tmp_path / "vault"
    res = runner.invoke(app, ["install", host_id, "--vault", str(vault)])
    assert res.exit_code == 0, res.output

    cfg_path = host.detect_path()  # expands ~ against $HOME
    assert cfg_path.exists(), f"{host_id}: no config written at {cfg_path}"
    data = json.loads(cfg_path.read_text())
    root_key = getattr(host, "root_key", None) or "mcpServers"
    servers = data[root_key]
    assert "agentcairn" in servers, f"{host_id}: no agentcairn entry"
    blob = json.dumps(servers["agentcairn"])
    assert "CAIRN_VAULT" in blob
    assert "CAIRN_INDEX" not in blob  # index is vault-derived
```

## Task 2 (Leg 4): CC plugin hook contract via PATH shim

**Files:** Create `tests/e2e/test_plugin_hooks.py`.

Context: `plugin/scripts/session-start.sh` and `session-end.sh` invoke `uvx --from agentcairn>=0.2 cairn …`. A PATH shim named `uvx` that drops the leading `--from <pkg>` and execs the local `cairn` (on PATH under `uv run`) makes the real scripts run against our code. `session-start.sh` takes the fast digest path only when `$HOME/.cache/agentcairn/indexes` exists; it then runs `cairn recent --vault <v> -n 5 --json` and emits `{"hookSpecificOutput":{"hookEventName":"SessionStart","additionalContext":…}}`. `session-end.sh` runs `cairn sweep --vault <v> --project <cwd>` detached (`nohup … &`).

- [ ] **Step 1: Shim helper + session-start test**

```python
# SPDX-License-Identifier: Apache-2.0
import json
import os
import subprocess
import time
from pathlib import Path

import pytest

from cairn.embed import FakeEmbedder
from cairn.index import build_fts, index_vault, open_index

REPO = Path(__file__).resolve().parents[2]
START = REPO / "plugin" / "scripts" / "session-start.sh"
END = REPO / "plugin" / "scripts" / "session-end.sh"


def _shim_dir(tmp_path: Path, body: str) -> Path:
    """Create a bin dir with a fake `uvx` that runs `body` (a shell snippet that
    receives the post-`uvx` argv). Real `uvx --from agentcairn>=0.2 cairn X` ->
    body sees `--from agentcairn>=0.2 cairn X`."""
    d = tmp_path / "bin"
    d.mkdir(exist_ok=True)
    uvx = d / "uvx"
    uvx.write_text("#!/bin/sh\n" + body + "\n")
    uvx.chmod(0o755)
    return d


def _seed_indexed_vault(tmp_path: Path) -> Path:
    vault = tmp_path / "vault"
    vault.mkdir()
    (vault / "a.md").write_text("---\ntitle: Pin DuckDB to 1.1\npermalink: pin-duckdb\n---\nWe pinned DuckDB.\n")
    (vault / "b.md").write_text("---\ntitle: Vault is the source of truth\npermalink: vault-truth\n---\nFiles are truth.\n")
    emb = FakeEmbedder(dim=8)
    # Build the vault-scoped index where `cairn recent` will look for it.
    # session-start checks ~/.cache/agentcairn/indexes; build the real index there.
    return vault


def test_session_start_emits_digest(tmp_path, monkeypatch):
    monkeypatch.setenv("HOME", str(tmp_path))
    vault = _seed_indexed_vault(tmp_path)
    # Reindex via the CLI so the vault-scoped index lands where `recent` reads it.
    # uvx shim -> drop `--from <pkg>`, exec local cairn.
    shim = _shim_dir(tmp_path, 'shift 2\nexec cairn "$@"')
    env = dict(os.environ, HOME=str(tmp_path), PATH=f"{shim}:{os.environ['PATH']}", CAIRN_EMBEDDER="fake")
    subprocess.run(["sh", "-c", f'cairn reindex "{vault}" --embedder fake'], env=env, check=True)
    # Ensure the fast-path marker dir exists.
    (tmp_path / ".cache" / "agentcairn" / "indexes").mkdir(parents=True, exist_ok=True)

    res = subprocess.run(["sh", str(START), str(vault)], env=env, stdin=subprocess.DEVNULL,
                         capture_output=True, text=True, timeout=60)
    assert res.returncode == 0, res.stderr
    out = res.stdout.strip()
    assert out, "session-start emitted nothing despite an indexed vault"
    payload = json.loads(out)
    assert payload["hookSpecificOutput"]["hookEventName"] == "SessionStart"
    assert "Pin DuckDB" in payload["hookSpecificOutput"]["additionalContext"]
```

Run: `uv run pytest tests/e2e/test_plugin_hooks.py::test_session_start_emits_digest -v`. Expected PASS. The reindex-location detail is the crux: `cairn recent --vault <v>` derives the vault-scoped index path from the vault under `~/.cache/agentcairn/indexes/<key>.duckdb`; running `cairn reindex <v>` with `HOME=tmp` writes it there, so the in-test `_seed_indexed_vault`'s manual `open_index` is unnecessary — REMOVE the unused `open_index/index_vault/build_fts` imports if you don't build the index inline (let the `cairn reindex` subprocess own it). If `recent` finds nothing, debug by running `cairn recent --vault <v> -n 5 --json` under the same env and inspect.

```python
def test_session_end_invokes_sweep(tmp_path, monkeypatch):
    monkeypatch.setenv("HOME", str(tmp_path))
    vault = tmp_path / "vault"
    vault.mkdir()
    rec = tmp_path / "argv.txt"
    # uvx shim: drop `--from <pkg>`, record the rest (the `cairn …` argv), exit 0.
    shim = _shim_dir(tmp_path, f'shift 2\nprintf "%s\\n" "$*" >> "{rec}"\nexit 0')
    env = dict(os.environ, HOME=str(tmp_path), PATH=f"{shim}:{os.environ['PATH']}")
    hook_json = '{"cwd": "/Users/x/proj"}'
    res = subprocess.run(["sh", str(END), str(vault)], input=hook_json, env=env,
                         capture_output=True, text=True, timeout=60)
    assert res.returncode == 0, res.stderr
    # session-end detaches the sweep; poll briefly for the recorded invocation.
    deadline = time.time() + 10
    recorded = ""
    while time.time() < deadline:
        if rec.exists():
            recorded = rec.read_text()
            if "sweep" in recorded:
                break
        time.sleep(0.2)
    assert "sweep" in recorded, f"session-end did not invoke sweep; recorded={recorded!r}"
    assert "--vault" in recorded and str(vault) in recorded
    assert "--project" in recorded and "/Users/x/proj" in recorded
```

```python
# SPDX-License-Identifier: Apache-2.0
import asyncio
import os
from pathlib import Path

import pytest

from cairn.embed import FakeEmbedder
from cairn.index import build_fts, index_vault, open_index

REPO = Path(__file__).resolve().parents[2]


def _seed_vault(tmp_path: Path) -> Path:
    vault = tmp_path / "vault"
    vault.mkdir()
    (vault / "note.md").write_text(
        "---\ntitle: Pin DuckDB to 1.1\npermalink: pin-duckdb\n---\nWe pinned DuckDB to 1.1.\n"
    )
    return vault


@pytest.mark.skipif(not os.environ.get("CAIRN_E2E"), reason="set CAIRN_E2E=1 for the stdio MCP contract test")
def test_mcp_stdio_contract(tmp_path):
    try:
        from mcp import ClientSession, StdioServerParameters
        from mcp.client.stdio import stdio_client
    except Exception as exc:
        pytest.skip(f"mcp client unavailable: {exc}")

    vault = _seed_vault(tmp_path)
    params = StdioServerParameters(
        command="uv",
        args=["run", "agentcairn"],
        env=dict(os.environ, CAIRN_VAULT=str(vault), CAIRN_EMBEDDER="fake"),
        cwd=str(REPO),
    )

    async def _run():
        async with stdio_client(params) as (read, write):
            async with ClientSession(read, write) as session:
                await session.initialize()
                tools = {t.name for t in (await session.list_tools()).tools}
                assert {"recall", "search", "recent", "build_context", "remember"} <= tools
                # call recall for the seeded note (confirm the arg name in server.py)
                result = await session.call_tool("recall", {"query": "why is DuckDB pinned"})
                text = " ".join(
                    getattr(c, "text", "") for c in result.content
                ).lower()
                assert "duckdb" in text
                return text

    out = asyncio.run(_run())
    assert out
```
